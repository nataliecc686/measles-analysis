Creating a function to extract total_aqi for each country

get_aqi <- function(city, country){
  
  base_url <- "https://api.api-ninjas.com/v1/airquality?"
  
  city <- paste0("city=", gsub(" ", "%20", city))
  
  country <- paste0("country=", gsub(" ", "%20", country))

  my_url <- glue("{base_url}{city}&{country}")

  my_key = "r28T1kQCHLHvPuqrJm8YnCgtmMyQJWiKcuYyuz9u"
  
  my_row <- GET(my_url, add_headers("X-Api-Key" = my_key))

  my_data <- fromJSON(rawToChar(my_row$content))
  
  if (length(my_data$overall_aqi) == 0){
    
    my_total_aqi <- NA
    
  }else {
    
    my_total_aqi <- my_data$overall_aqi
    
  }
  
  return(my_total_aqi)
  
}
